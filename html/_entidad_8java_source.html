591  private PilaDeTiles caminoMasCorto(final int xInicial, final int yInicial, final int xFinal, final int yFinal) {
592  Grafo grafoLibres = mundo.obtenerGrafoDeTilesNoSolidos();
593  // Transformo las coordenadas iniciales y finales en indices
594  int nodoInicial = (yInicial - grafoLibres.obtenerNodos()[0].obtenerY())
595  * (int) Math.sqrt(grafoLibres.obtenerCantidadDeNodosTotal()) + xInicial
596  - grafoLibres.obtenerNodos()[0].obtenerX();
597 
598  int nodoFinal = (yFinal - grafoLibres.obtenerNodos()[0].obtenerY())
599  * (int) Math.sqrt(grafoLibres.obtenerCantidadDeNodosTotal()) + xFinal
600  - grafoLibres.obtenerNodos()[0].obtenerX();
601 
602  // Hago todo
603  double[] vecCostos = new double[grafoLibres.obtenerCantidadDeNodosTotal()];
604  int[] vecPredecesores = new int[grafoLibres.obtenerCantidadDeNodosTotal()];
605  boolean[] conjSolucion = new boolean[grafoLibres.obtenerCantidadDeNodosTotal()];
606  int cantSolucion = 0;
607  // Lleno la matriz de costos de numeros grandes
608  for (int i = 0; i < grafoLibres.obtenerCantidadDeNodosTotal(); i++) {
609  vecCostos[i] = Double.MAX_VALUE;
610  }
611  // Adyacentes al nodo inicial
612  conjSolucion[nodoInicial] = true;
613  cantSolucion++;
