536  private PilaDeTiles caminoMasCorto(final int xInicial, final int yInicial, final int xFinal, final int yFinal) {
537  Grafo grafoLibres = mundo.obtenerGrafoDeTilesNoSolidos();
538  // Transformo las coordenadas iniciales y finales en indices
539  int nodoInicial = (yInicial - grafoLibres.obtenerNodos()[0].obtenerY())
540  * (int) Math.sqrt(grafoLibres.obtenerCantidadDeNodosTotal()) + xInicial
541  - grafoLibres.obtenerNodos()[0].obtenerX();
542 
543  int nodoFinal = (yFinal - grafoLibres.obtenerNodos()[0].obtenerY())
544  * (int) Math.sqrt(grafoLibres.obtenerCantidadDeNodosTotal()) + xFinal
545  - grafoLibres.obtenerNodos()[0].obtenerX();
546 
547  // Hago todo
548  double[] vecCostos = new double[grafoLibres.obtenerCantidadDeNodosTotal()];
549  int[] vecPredecesores = new int[grafoLibres.obtenerCantidadDeNodosTotal()];
550  boolean[] conjSolucion = new boolean[grafoLibres.obtenerCantidadDeNodosTotal()];
551  int cantSolucion = 0;
552  // Lleno la matriz de costos de numeros grandes
553  for (int i = 0; i < grafoLibres.obtenerCantidadDeNodosTotal(); i++) {
554  vecCostos[i] = Double.MAX_VALUE;
555  }
556  // Adyacentes al nodo inicial
557  conjSolucion[nodoInicial] = true;
558  cantSolucion++;
559  vecCostos[nodoInicial] = 0;
560  Nodo[] adyacentes = grafoLibres.obtenerNodos()[nodoInicial].obtenerNodosAdyacentes();
561  for (int i = 0; i < grafoLibres.obtenerNodos()[nodoInicial].obtenerCantidadDeAdyacentes(); i++) {
562  if (estanEnDiagonal(grafoLibres.obtenerNodos()[nodoInicial],
563  grafoLibres.obtenerNodos()[adyacentes[i].obtenerIndice()])) {
564  vecCostos[adyacentes[i].obtenerIndice()] = 1.5;
565  } else {
566  vecCostos[adyacentes[i].obtenerIndice()] = 1;
567  }
568  vecPredecesores[adyacentes[i].obtenerIndice()] = nodoInicial;
569  }
570  // Aplico Dijkstra
571  while (cantSolucion < grafoLibres.obtenerCantidadDeNodosTotal()) {
572  // Elijo W perteneciente al conjunto restante tal que el costo de W
573  // sea minimo
574  double minimo = Double.MAX_VALUE;
575  int indiceMinimo = 0;
576  Nodo nodoW = null;
577  for (int i = 0; i < grafoLibres.obtenerCantidadDeNodosTotal(); i++) {
578  if (!conjSolucion[i] && vecCostos[i] < minimo) {
579  nodoW = grafoLibres.obtenerNodos()[i];
580  minimo = vecCostos[i];
581  indiceMinimo = i;
582  }
583  }
584  // Pongo a W en el conj solucion
